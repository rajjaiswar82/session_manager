```bash
setup.bat
run.bat

# Manual
python -m venv venv
venv\Scripts\activate
pip install -r requirements.txt
uvicorn app.main:app --reload
```

### Testing
```bash
# Automated tests
python test_api.py

# Manual server start
uvicorn app.main:app --reload

```

### Database
```bash
# Create database
psql -U postgres -c "CREATE DATABASE session_manager;"

# Run setup script

psql -U postgres -d session_manager
```
